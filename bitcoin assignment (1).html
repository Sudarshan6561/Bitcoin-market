import numpy as np
import random
from datetime import datetime, timedelta

# Generate 30 days of data
dates = [datetime.now() - timedelta(days=i) for i in range(30)]
open_prices = np.random.uniform(low=30000, high=60000, size=30)
high_prices = open_prices + np.random.uniform(low=100, high=1000, size=30)
low_prices = open_prices - np.random.uniform(low=100, high=1000, size=30)
close_prices = np.random.uniform(low=30000, high=60000, size=30)
volume = np.random.uniform(low=5000, high=100000, size=30)
market_cap = np.random.uniform(low=500000000, high=1000000000, size=30)

# Create the DataFrame
bitcoin_df['Date'] = dates
bitcoin_df['Open'] = open_prices
bitcoin_df['High'] = high_prices
bitcoin_df['Low'] = low_prices
bitcoin_df['Close'] = close_prices
bitcoin_df['Volume'] = volume
bitcoin_df['Market Cap'] = market_cap

# View the first few rows
print(bitcoin_df.head())
